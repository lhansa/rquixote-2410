eble::dim_ret
function (x, ..., option = c("rect", "adbug", "hill", "atan", 
    "inversec", "asymmetrics", "asymptotic"), rescale = FALSE) 
{
    option = match.arg(option)
    params = suppressWarnings(vapply(list(...), as.numeric, numeric(1L)))
    if (suppressWarnings(any(is.na(params)))) {
        if (list(...)[which(lapply(list(...), is.character) == 
            TRUE)][[1]] %in% c("rect", "adbug", "hill", "atan", 
            "inversec", "asymmetrics", "asymptotic")) {
            option = list(...)[which(lapply(list(...), is.character) == 
                TRUE)][[1]]
            params = params[!is.na(params)]
        }
        else {
            stop("check your dim_ret arguments! Specify the option argument, e.g. option = 'rect'")
        }
    }
    if (length(params) != (n <- switch(option, rect = , atan = 1, 
        adbug = , hill = 2, inversec = 1, asymmetrics = 2, asymptotic = 1))) {
        if (option %in% c("hill", "adbug", "asymmetrics")) {
            stop("s parameter value was not provided. Hill, adbug and asymmetrics functions need two parameter values.")
        }
        if (length(params) == 2 & option %in% c("rect")) {
            stop("option value was not provided. s parameter applies only to hill, adbug and asymmetrics functions.")
        }
        else {
            stop(sprintf("incorrect number of parameters for option '%s' in dim ret: expected %i, but got %i", 
                option, n, length(params)))
        }
    }
    if (any(!is.finite(params))) {
        stop("non-numeric parameter(s) in dim ret")
    }
    if (class(x) == "integer" && class(params) == "integer") {
        params = as.double(params)
    }
    ind = params == Inf
    if (all(ind)) {
        return(x)
    }
    if (any(params < 0)) {
        stop("Dimishing returns are not defined for negative values.")
    }
    if (length(params) == 2) {
        if (params[2] == 0) {
            stop("Dimishing returns are not defined for zero s value.")
        }
        if (max(x, params[1], na.rm = TRUE)^params[2] == Inf & 
            option == "hill") {
            params[2] = 1023 * (log(2)/log(max(x, params[1], 
                na.rm = TRUE)))
            stop(sprintf("the selected s parameter is causing overflow error. The maximum finite value is %s", 
                round(params[2], 2)))
        }
        if (max(x, na.rm = TRUE)^params[2] == Inf & option == 
            "adbug") {
            params[2] = 1023 * (log(2)/log(max(x, na.rm = TRUE)))
            stop(sprintf("the selected s parameter is causing overflow error. The maximum finite value is %s", 
                round(params[2], 2)))
        }
    }
    y = switch(option, rect = x/(x + params[1]), adbug = x^params[2]/(x^params[2] + 
        params[1]), hill = x^params[2]/(x^params[2] + params[1]^params[2]), 
        atan = atan(x/params[1]), inversec = 1 - exp(-x/params[1]), 
        asymmetrics = 1/(1 + exp(-params[1] * (x - params[2]))), 
        asymptotic = 1 - exp(-x/params[1]))
    if (rescale) {
        y = y * params[1]
    }
    if (any(ind)) {
        y[ind] = x[ind]
    }
    return(y)
}

eble::adstock
function (x, p, normalize = TRUE, reverse = FALSE) 
{
    if (length(p) > 1) {
        stop("Input p to adstock must have a length of 1.")
    }
    if (p < 0 | p >= 1) {
        stop("Input p to adstock must be in [0, 1)")
    }
    if (class(x) == "factor") {
        warning("factor supplied to adstock function")
        x = as.character(x)
    }
    x[is.na(x)] = 0
    if (reverse == FALSE) {
        if (normalize) {
            x = x * (1 - p)
        }
        x = stats::filter(x, p, method = "recursive")
        return(as.numeric(x))
    }
    else {
        x = rev(adstock(rev(x), p, normalize))
        return(as.numeric(x))
    }
}
